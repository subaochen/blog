\def\@xxNoOpt#1{arg is #1}

    \futurelet\xxLookedAtToken\xxDecide
}

\def\xxDecide{
    \ifx\xxLookedAtToken [
        \let\next = \@xxWithOpt
    \else
        \let\next = \@xxNoOpt
    \fi
    \next
}
